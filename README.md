        function openTab(evt, tabName) {
            let i, tabcontent, tablinks;
            tabcontent = document.getElementsByClassName("tab-content");
            for (i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; }
            tablinks = document.getElementsByClassName("tab-button");
            for (i = 0; i < tablinks.length; i++) { tablinks[i].className = tablinks[i].className.replace(" active", ""); }
            document.getElementById(tabName).style.display = "block";
            evt.currentTarget.className += " active";
            
            const typeMap = { 
                'VerbsTab': 'verbs', 
                'IrregularVerbsTab': 'irregular', 
                'PhrasalVerbsTab': 'phrasal', 
                'VocabularyTab': 'vocab' 
            };
            flashcards.displayCard(typeMap[tabName]);
        }

        const flashcards = {
            data: { verbs: [], irregular: [], phrasal: [], vocab: [] },
            state: {
                verbs: { currentIndex: 0, isRevealed: false },
                irregular: { currentIndex: 0, isRevealed: false },
                phrasal: { currentIndex: 0, isRevealed: false },
                vocab: { currentIndex: 0, isRevealed: false }
            },

            shuffleArray: function(array) { 
                for (let i = array.length - 1; i > 0; i--) {
                    const j = Math.floor(Math.random() * (i + 1));
                    [array[i], array[j]] = [array[j], array[i]];
                }
                return array;
            },

            loadData: function() {
                try {
                    this.data.verbs = this.shuffleArray(JSON.parse(document.getElementById('regularVerbsData').textContent));
                    this.data.irregular = this.shuffleArray(JSON.parse(document.getElementById('irregularVerbsData').textContent));
                    this.data.phrasal = this.shuffleArray(JSON.parse(document.getElementById('phrasalVerbsData').textContent));
                    this.data.vocab = this.shuffleArray(JSON.parse(document.getElementById('vocabularyData').textContent));
                    console.log("Datos cargados y barajados:", this.data);
                } catch (e) { console.error("Error cargando/barajando datos JSON:", e); }
            },

            getElement: function(type, elementIdPart) { return document.getElementById(type + "_" + elementIdPart); },

            displayCard: function(type) {
                const currentData = this.data[type];
                const currentState = this.state[type];
                console.log(`Displaying card for type: ${type}, index: ${currentState.currentIndex}`);

                const cardInfoEl = this.getElement(type, 'cardInfo');
                const frontEl = this.getElement(type, 'front');
                const backEl = this.getElement(type, 'back');
                const revealBtnEl = this.getElement(type, 'revealBtn');
                const prevBtnEl = this.getElement(type, 'prevBtn');
                const nextBtnEl = this.getElement(type, 'nextBtn');

                if (!currentData || currentData.length === 0) {
                    if(cardInfoEl) cardInfoEl.textContent = "No hay datos para mostrar.";
                    if(frontEl) frontEl.textContent = "N/A"; // Default to N/A if no data
                    if(backEl) backEl.style.display = 'none';
                    if(revealBtnEl) revealBtnEl.style.display = 'none';
                    if(prevBtnEl) { prevBtnEl.style.display = 'inline-block'; prevBtnEl.disabled = true; }
                    if(nextBtnEl) { nextBtnEl.style.display = 'inline-block'; nextBtnEl.disabled = true; }
                    return;
                }
                
                currentState.isRevealed = false;
                if(backEl) backEl.style.display = 'none';
                if(revealBtnEl) { revealBtnEl.textContent = 'Revelar Reverso'; revealBtnEl.style.display = 'inline-block'; }
                if(prevBtnEl) prevBtnEl.style.display = 'inline-block';
                if(nextBtnEl) nextBtnEl.style.display = 'inline-block';

                const card = currentData[currentState.currentIndex];
                if (!card) { 
                    console.error(`Error: Tarjeta no encontrada para ${type} en índice ${currentState.currentIndex}`); 
                    if(frontEl) frontEl.textContent = "Error";
                    return; 
                }

                // Default front text for verbs and vocab
                frontEl.textContent = card.front_text || "N/A";

                if (type === 'verbs' || type === 'vocab') {
                    // frontEl.textContent is already set by the general rule above if front_text exists
                    this.getElement(type, 'translation').textContent = card.back_text || "N/A";
                    if(this.getElement(type, 'exampleEnglish')) this.getElement(type, 'exampleEnglish').textContent = card.example_english || "[Ej. no disp.]";
                    if(this.getElement(type, 'exampleSpanish')) this.getElement(type, 'exampleSpanish').textContent = card.example_spanish || "[Ej. no disp.]";
                } else if (type === 'phrasal') {
                    // *** Specific override for phrasal front text ***
                    frontEl.textContent = card.base_verb || "N/A"; 
                    
                    this.getElement(type, 'preposition').textContent = card.preposition_part || "[Prep. no disp.]";
                    this.getElement(type, 'translation_spa').textContent = card.translation_spa || "[Trad. no disp.]";
                    if(this.getElement(type, 'exampleEnglish')) this.getElement(type, 'exampleEnglish').textContent = card.example_english || "[Ej. no disp.]";
                    if(this.getElement(type, 'exampleSpanish')) this.getElement(type, 'exampleSpanish').textContent = card.example_spanish || "[Ej. no disp.]";
                } else if (type === 'irregular') {
                    // frontEl.textContent is already set by the general rule above if front_text exists
                    this.getElement(type, 'translation').textContent = card.translation || "N/A";
                    this.getElement(type, 'past_simple').textContent = card.past_simple || "N/A";
                    this.getElement(type, 'past_participle').textContent = card.past_participle || "N/A";
                }
                
                if(cardInfoEl) cardInfoEl.textContent = `Flashcard ${currentState.currentIndex + 1} de ${currentData.length}`;
                if(prevBtnEl) prevBtnEl.disabled = currentState.currentIndex === 0;
                if(nextBtnEl) nextBtnEl.disabled = currentState.currentIndex === currentData.length - 1;
            },

            toggleReveal: function(type) {
                const currentState = this.state[type];
                if (!this.data[type] || this.data[type].length === 0) return;
                currentState.isRevealed = !currentState.isRevealed;
                this.getElement(type, 'back').style.display = currentState.isRevealed ? 'block' : 'none';
                this.getElement(type, 'revealBtn').textContent = currentState.isRevealed ? 'Ocultar Reverso' : 'Revelar Reverso';
            },

            nextCard: function(type) {
                const currentState = this.state[type];
                if (currentState.currentIndex < this.data[type].length - 1) {
                    currentState.currentIndex++;
                    this.displayCard(type);
                }
            },

            prevCard: function(type) {
                const currentState = this.state[type];
                if (currentState.currentIndex > 0) {
                    currentState.currentIndex--;
                    this.displayCard(type);
                }
            },

            init: function() {
                this.loadData();
                // This ensures the DOM is fully loaded before trying to access elements for the initial tab.
                document.addEventListener('DOMContentLoaded', () => {
                    const initialTabButton = document.querySelector('.tab-button.active');
                    if (initialTabButton) {
                        const initialTabName = initialTabButton.getAttribute('onclick').match(/'([^']+)'/)[1];
                        const typeMap = { 'VerbsTab': 'verbs', 'IrregularVerbsTab': 'irregular', 'PhrasalVerbsTab': 'phrasal', 'VocabularyTab': 'vocab' };
                        // Call displayCard directly for the initial active tab's type AFTER data is loaded.
                        // The openTab function also calls displayCard, but this ensures the very first display is correct.
                        this.displayCard(typeMap[initialTabName]); 
                    } else {
                        // Fallback if no tab is marked active (shouldn't happen with current HTML)
                        // but good for robustness.
                        const firstButton = document.querySelector('.tab-button');
                        if (firstButton) {
                            const firstTabName = firstButton.getAttribute('onclick').match(/'([^']+)'/)[1];
                            // Manually trigger the openTab for the first button if none are active.
                            openTab({ currentTarget: firstButton }, firstTabName);
                        }
                    }
                });
            }
        };
        
        flashcards.init();

    </script>
